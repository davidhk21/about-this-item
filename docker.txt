-- build the docker container with the name of the container
docker build -t my-cassandra .

-- run the docker container
-- file path to left of : is the file path on local computer
-- file path after the : is file path from the image online
docker run -d -p 9042:9042 -v /data/my-cassandra:/var/lib/cassandra my-cassandra

-- get into the cqlsh shell for specific containerID
docker exec -ti 79c93d5d0340 cqlsh localhost;

-- see stats for specific container
docker stats container id

-- access the root of the container
docker exec -ti 73a13781c717 ls


-- NOTES
-- check containers
docker container ls
-- check images
docker images
